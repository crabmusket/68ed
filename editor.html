****************************************
*   -- Sample Program --
*   This program writes a series of
*   characters from a phrase to PORTB
*****************************************
REGBASE equ     $1000
PORTB   equ     $04

        org     $0000

init:
        ldx     #phrase
        ldy     #REGBASE
loop:
        ldaa    0,x
        beq     init
        staa    PORTB,y
        inx
        jmp     loop
phrase:
        fcb     "Hello, world!"
